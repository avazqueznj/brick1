#include <Arduino.h>
#include <Wire.h>
#include <SPI.h>

#include <ArduCAM.h>
#include "memorysaver.h"   // OV2640 must be enabled here

// ================= CONFIG =================
static constexpr int CAM_CS = 10;     // change if needed
static const size_t JPG_MAX = 256 * 1024;
// ==========================================

ArduCAM* cam = nullptr;
uint8_t* jpg = nullptr;
size_t jpg_len = 0;

// ---------- helpers ----------
static void die(const char* msg) {
  Serial.println(msg);
  Serial.flush();
  while (1) {
    digitalWrite(LED_BUILTIN, !digitalRead(LED_BUILTIN));
    delay(300);
  }
}

static void dump_bytes(const uint8_t* p, size_t n) {
  for (size_t i = 0; i < n; i++) {
    if (p[i] < 16) Serial.print("0");
    Serial.print(p[i], HEX);
    Serial.print(" ");
  }
  Serial.println();
}

static void read_fifo(uint32_t len) {
  if (len == 0 || len == 0xFFFFFFFF) die("[CAM] FIFO length invalid");
  if (len > JPG_MAX) die("[CAM] FIFO > buffer");

  cam->CS_LOW();
  SPI.transfer(BURST_FIFO_READ);
  for (uint32_t i = 0; i < len; i++) {
    jpg[i] = SPI.transfer(0x00);
  }
  cam->CS_HIGH();

  jpg_len = (size_t)len;
}

static void take_picture() {
  Serial.println("\n[CAM] Capture start");
  Serial.flush();

  cam->flush_fifo();
  cam->clear_fifo_flag();
  cam->start_capture();

  uint32_t t0 = millis();
  while (!cam->get_bit(ARDUCHIP_TRIG, CAP_DONE_MASK)) {
    if (millis() - t0 > 6000) die("[CAM] Capture timeout");
    delay(2);
  }

  uint32_t len = cam->read_fifo_length();
  Serial.print("[CAM] FIFO length = ");
  Serial.println(len);

  read_fifo(len);

  Serial.print("[CAM] SOI: ");
  Serial.print(jpg[0], HEX); Serial.print(" ");
  Serial.println(jpg[1], HEX);

  Serial.print("[CAM] EOI: ");
  Serial.print(jpg[jpg_len - 2], HEX); Serial.print(" ");
  Serial.println(jpg[jpg_len - 1], HEX);

  Serial.println("[CAM] First 32 bytes:");
  dump_bytes(jpg, 32);

  Serial.println("[CAM] Last 16 bytes:");
  dump_bytes(jpg + jpg_len - 16, 16);

  Serial.flush();
}

// ---------- Arduino ----------
void setup() {
  pinMode(LED_BUILTIN, OUTPUT);
  digitalWrite(LED_BUILTIN, LOW);

  // Give terminal time to attach
  delay(3000);

  Serial.begin(9600);
  delay(200);
  Serial.flush();

  Serial.println("=======================================");
  Serial.println(" ArduCAM Mini (B0067) GIGA JPEG TEST ");
  Serial.println("=======================================");

  jpg = (uint8_t*)malloc(JPG_MAX);
  if (!jpg) die("[FATAL] malloc failed");

  Serial.println("[BOOT] Buffer OK");

  SPI.begin();
  Wire.begin();
  Serial.println("[BOOT] SPI + I2C OK");

  pinMode(CAM_CS, OUTPUT);
  digitalWrite(CAM_CS, HIGH);

  cam = new ArduCAM(OV2640, CAM_CS);
  Serial.println("[BOOT] Camera object created");

  // cam->write_reg(ARDUCHIP_TEST1, 0x55);
  // uint8_t v = cam->read_reg(ARDUCHIP_TEST1);
  // Serial.print("[BOOT] SPI TEST1 = 0x");
  // Serial.println(v, HEX);
  // if (v != 0x55) die("[BOOT] SPI test failed");

  cam->set_format(JPEG);
  cam->InitCAM();
  cam->OV2640_set_JPEG_size(OV2640_640x480);
  delay(200);

  Serial.println("[BOOT] Camera init OK");
  Serial.flush();
}

void loop() {
  // heartbeat
  digitalWrite(LED_BUILTIN, !digitalRead(LED_BUILTIN));

  take_picture();

  delay(3000);
}
